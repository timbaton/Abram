  name varchar(35),

  password varchar(50)

create table desk (
  id serial primary key,
  name varchar(35),
  data_of_creation timestamp
);

create table user_to_table(
  id serial not null,
  user_id integer references "user"(id) unique ,
  desk_id integer references desk(id) unique
);
